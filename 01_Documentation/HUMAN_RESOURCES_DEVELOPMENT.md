```python
# Performance Management System
class PerformanceManagement:
    def __init__(self):
        self.okr_framework = {
            'company_okrs': 'Quarterly strategic objectives',
            'team_okrs': 'Team-specific goals aligned with company',
            'individual_okrs': 'Personal goals aligned with team/company',
            'review_cycle': 'Quarterly check-ins + annual review'
        }
    
    def performance_dimensions(self):
        return {
            'results_delivery': 0.4,      # What you achieve
            'behavior_values': 0.3,       # How you achieve it
            'growth_learning': 0.2,       # How you develop
            'collaboration': 0.1          # How you work with others
        }
    
    def rating_scale(self):
        return {
            'exceptional': 5,    # Top 10% - Exceeds all expectations
            'exceeds': 4,        # 20% - Exceeds most expectations
            'meets': 3,          # 60% - Meets all expectations
            'developing': 2,     # 10% - Developing, needs support
            'underperforming': 1 # <5% - Performance improvement needed
        }
```
