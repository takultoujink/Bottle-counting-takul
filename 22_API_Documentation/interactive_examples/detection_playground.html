import requests # Using our Python SDK from object_detection_sdk import APIClient client = APIClient(api_key="YOUR_API_KEY") # Upload and detect with open("image.jpg", "rb") as f: results = client.upload_image(f, model="yolo", confidence=0.5) print(f"Found {len(results['detections'])} objects") for detection in results['detections']: print(f"- {detection['class']}: {detection['confidence']:.2f}")
